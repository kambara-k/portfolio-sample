Oracle SQLの基本コマンド

Oracle SQLでは、基本的なデータ操作として SELECT、INSERT、UPDATE、DELETEなどを使用する。 

						<-- データ取得（検索） -->
						SELECT * FROM テーブル名;
						
						<-- 条件付き検索 -->
						SELECT * FROM テーブル名
						WHERE 列名 = '値';
						
						<-- 並び替え -->
						SELECT * FROM テーブル名
						ORDER BY 列名 ASC;
						
						<-- データ追加 -->
						INSERT INTO テーブル名 (列1, 列2)
						VALUES ('値1', '値2');
						
						<-- データ更新 -->
						UPDATE テーブル名
						SET 列1 = '値'
						WHERE 条件;
						
						<-- データ削除 -->
						DELETE FROM テーブル名
						WHERE 条件;
						
						<-- テーブル作成 -->
						CREATE TABLE テーブル名 (
						    列1 データ型,
						    列2 データ型
						);

さらに重要な応用として、複数テーブルの結合や集計処理もよく使われる。 

						<-- テーブル結合（JOIN） -->
						SELECT *
						FROM テーブルA
						JOIN テーブルB
						ON テーブルA.id = テーブルB.id;
						
						<-- 集計（GROUP BY） -->
						SELECT 列名, COUNT(*)
						FROM テーブル名
						GROUP BY 列名;
